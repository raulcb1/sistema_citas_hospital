// Procesar formulario
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $servicio_id = $_POST['servicio_id'];
    $dias = $_POST['dias'] ?? [];
    $hora_inicio = $_POST['hora_inicio'];
    $hora_fin = $_POST['hora_fin'];
    $intervalo = $_POST['intervalo'];
    $capacidad = $_POST['capacidad'];
    $activo = isset($_POST['activo']) ? 1 : 0;

    // Validar datos
    $errores = [];
    
    if (empty($dias)) {
        $errores[] = "Seleccione al menos un día";
    }
    
    if (strtotime($hora_fin) <= strtotime($hora_inicio)) {
        $errores[] = "La hora final debe ser mayor a la hora inicial";
    }
    
    if (empty($errores)) {
        // Crear string de días
        $dias_semana = implode(',', $dias);
        
        // Insertar en base de datos
        $stmt = $conn->prepare("
            INSERT INTO horarios_servicio (
                servicio_id, 
                dias_semana, 
                hora_inicio, 
                hora_fin, 
                intervalo_min, 
                capacidad_por_intervalo, 
                activo
            ) VALUES (?, ?, ?, ?, ?, ?, ?)
        ");
        
        $stmt->bind_param(
            "isssiii",
            $servicio_id,
            $dias_semana,
            $hora_inicio,
            $hora_fin,
            $intervalo,
            $capacidad,
            $activo
        );
        
        if ($stmt->execute()) {
            $mensaje = "Horario guardado exitosamente!";
        } else {
            $error = "Error al guardar: " . $conn->error;
        }
    }
}

");

// Mejorar usabilidad de inputs de tiempo
document.querySelectorAll('input[type="time"]').forEach(input => {
    input.addEventListener('change', (e) => {
        const time = e.target.value;
        const [hours, minutes] = time.split(':');
        const roundedMinutes = Math.floor(minutes / 5) * 5;
        e.target.value = `${hours}:${roundedMinutes.toString().padStart(2, '0')}`;
    });
});
